int gcd(int a, int b)
{
    int g;
    int smaller = a>b?b:a;
    for(int i=smaller;i>=1;i--)
    {
        if(a%i==0 && b%i==0)
        {
            g = i;
            break;
        }
    }
    return g;
}

{

    }

        return 0;

}
